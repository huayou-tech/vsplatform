std::string CStringHelper::UnicodeToUtf8(const std::wstring& wstr)
{
    if (wstr.empty()) return std::string();
    
    int size_needed = WideCharToMultiByte(CP_UTF8, 0, &wstr[0], (int)wstr.size(), 
                                         NULL, 0, NULL, NULL);
    std::string strTo(size_needed, 0);
    WideCharToMultiByte(CP_UTF8, 0, &wstr[0], (int)wstr.size(), 
                       &strTo[0], size_needed, NULL, NULL);
    return strTo;
}

// 递归创建目录实现
bool CFileHelper::CreateDirectoryRecursive(const std::string& path)
{
    if (path.empty()) return false;
    
    std::string normalizedPath = path;
    // 标准化路径分隔符
    std::replace(normalizedPath.begin(), normalizedPath.end(), '/', '\\');
    
    // 如果已经存在，直接返回成功
    if (DirectoryExists(normalizedPath)) {
        return true;
    }
    
    // 递归创建父目录
    size_t pos = normalizedPath.find_last_of('\\');
    if (pos != std::string::npos && pos > 0) {
        std::string parentPath = normalizedPath.substr(0, pos);
        if (!CreateDirectoryRecursive(parentPath)) {
            return false;
        }
    }
    
    // 创建当前目录
    return ::CreateDirectoryA(normalizedPath.c_str(), NULL) || 
           GetLastError() == ERROR_ALREADY_EXISTS;
}

std::string CTimeHelper::GetCurrentTimeString(const std::string& format)
{
    auto now = std::chrono::system_clock::now();
    auto time_t = std::chrono::system_clock::to_time_t(now);
    
    struct tm local_tm;
    localtime_s(&local_tm, &time_t);
    
    char buffer[100];
    strftime(buffer, sizeof(buffer), format.c_str(), &local_tm);
    return std::string(buffer);
}

// 配置值获取模板实现
template<typename T>
T CConfigManager::GetValue(const std::string& key, const T& defaultValue)
{
    std::lock_guard<std::mutex> lock(m_ConfigMutex);
    
    std::vector<std::string> path = SplitKeyPath(key);
    Json::Value* currentNode = &m_ConfigRoot;
    
    for (const auto& part : path) {
        if (!currentNode->isObject() || !currentNode->isMember(part)) {
            return defaultValue;
        }
        currentNode = &(*currentNode)[part];
    }
    
    return JsonToValue<T>(*currentNode, defaultValue);
}

// JSON到C++类型的转换
template<typename T>
T JsonToValue(const Json::Value& jsonValue, const T& defaultValue)
{
    if constexpr (std::is_same_v<T, std::string>) {
        return jsonValue.isString() ? jsonValue.asString() : defaultValue;
    }
    else if constexpr (std::is_arithmetic_v<T>) {
        return jsonValue.isNumeric() ? static_cast<T>(jsonValue.asDouble()) : defaultValue;
    }
    else if constexpr (std::is_same_v<T, bool>) {
        return jsonValue.isBool() ? jsonValue.asBool() : defaultValue;
    }
    else {
        return defaultValue;
    }
}

template<typename T>
class ThreadSafeSingleton
{
private:
    static std::unique_ptr<T> m_Instance;
    static std::once_flag m_OnceFlag;
    
public:
    static T& Instance()
    {
        std::call_once(m_OnceFlag, []() {
            m_Instance.reset(new T());
        });
        return *m_Instance;
    }
    
    static void Destroy()

    }
};
